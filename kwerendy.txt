SELECT DISTINCT `pochodzenie` FROM `smok` ORDER BY pochodzenie ASC;
SELECT `rok`, AVG(smok.dlugosc) "Średnia długość" FROM `parada`, smok, udzial WHERE udzial.id_parada = parada.id AND udzial.id_smok = smok.id AND parada.rok > 2005 GROUP BY parada.rok;
